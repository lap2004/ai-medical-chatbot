# Medical AI Chatbot Database Schema v012026

## Bảng users_012026 (Quản lý người dùng)

CREATE TABLE users_012026 (
    id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    email VARCHAR(255) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    full_name VARCHAR(255),
    is_admin BOOLEAN NOT NULL DEFAULT FALSE,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    force_password_change BOOLEAN NOT NULL DEFAULT FALSE,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

## Bảng medical_012026 (Kho tri thức y tế)

CREATE TABLE medical_012026 (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title TEXT NOT NULL,
    question TEXT NOT NULL,
    answer TEXT NOT NULL,
    symptoms TEXT,
    treatment TEXT,
    embedding vector(1024)
);

## Bảng conversations_012026 (Phiên hội thoại)

CREATE TABLE conversations_012026 (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id INTEGER NOT NULL REFERENCES users_012026(id) ON DELETE CASCADE,

    title TEXT NOT NULL DEFAULT 'New chat',

    deleted_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    last_message_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_conversations_user_012026
    ON conversations_012026(user_id);

CREATE INDEX idx_conversations_user_active_012026
    ON conversations_012026(user_id, last_message_at DESC)
    WHERE deleted_at IS NULL;

## Bảng messages_012026 (Lịch sử tin nhắn)

CREATE TABLE messages_012026 (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    conversation_id UUID NOT NULL REFERENCES conversations_012026(id) ON DELETE CASCADE,
    user_id INTEGER REFERENCES users_012026(id) ON DELETE SET NULL,

    role message_role NOT NULL,
    content TEXT NOT NULL,

    model TEXT,
    prompt_tokens INTEGER,
    completion_tokens INTEGER,
    total_tokens INTEGER,

    parent_message_id UUID REFERENCES messages_012026(id) ON DELETE SET NULL,

    rag_context JSONB,

    deleted_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_messages_conversation_012026
    ON messages_012026(conversation_id, created_at ASC)
    WHERE deleted_at IS NULL;

CREATE INDEX idx_messages_parent_012026
    ON messages_012026(parent_message_id);

## Bảng message_feedback_012026 (Phản hồi chất lượng)

CREATE TABLE message_feedback_012026 (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    message_id UUID NOT NULL REFERENCES messages_012026(id) ON DELETE CASCADE,
    user_id INTEGER NOT NULL REFERENCES users_012026(id) ON DELETE CASCADE,

    value feedback_value NOT NULL,
    reason TEXT,

    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

    CONSTRAINT uq_message_feedback_012026 UNIQUE (message_id, user_id)
);

CREATE INDEX idx_feedback_message_012026
    ON message_feedback_012026(message_id);

CREATE TABLE message_reports_012026 (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    reporter_user_id INTEGER NOT NULL REFERENCES users_012026(id) ON DELETE CASCADE,

## Bảng message_citations_012026 (Trích dẫn)
