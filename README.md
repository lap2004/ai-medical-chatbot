## cấu trúc json

"id": UUID, duy nhất cho mỗi bản ghi, tương ứng với id UUID PRIMARY KEY DEFAULT gen_random_uuid() trong SQL.
"title": Tên loại bệnh, tương ứng với cột title TEXT NOT NULL.
"question": Câu hỏi về bệnh, tương ứng với cột question TEXT NOT NULL.
"answer": Câu trả lời về bệnh, tương ứng với cột answer TEXT NOT NULL.
"symptoms": Triệu chứng hoặc dấu hiệu nhận biết bệnh, tương ứng với symptoms TEXT.
"treatment": Phương pháp điều trị, tương ứng với treatment TEXT.

-- Bật các extension cần thiết
CREATE EXTENSION IF NOT EXISTS "pgcrypto";
CREATE EXTENSION IF NOT EXISTS vector;

-- Tạo bảng diseases
CREATE TABLE medical (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title TEXT NOT NULL,
    question TEXT NOT NULL,
    answer TEXT NOT NULL,
    symptoms TEXT,
    treatment TEXT,
    embedding vector(1024) -- vector 1024 chiều
);

CREATE TABLE users (
    id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    email VARCHAR(255) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    full_name VARCHAR(255),
    is_admin BOOLEAN NOT NULL DEFAULT FALSE,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    force_password_change BOOLEAN NOT NULL DEFAULT FALSE,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE chat_messages (
  id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  user_id INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  question TEXT NOT NULL,
  answer TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_chat_messages_user_time ON chat_messages(user_id, created_at DESC);

create extension if not exists pgcrypto;

create table if not exists users_30012026 (
    id integer generated by default as identity primary key,
    email varchar(255) not null unique,
    password varchar(255) not null,
    full_name varchar(255),
    is_admin boolean not null default false,
    is_active boolean not null default true,
    force_password_change BOOLEAN NOT NULL DEFAULT FALSE,
    created_at timestamptz not null default now()
);

create table conversations_30012026 (
  id              uuid primary key default gen_random_uuid(),
  user_id         integer not null references users(id) on delete cascade,

  title           text not null default 'New chat',

  -- active|archived|deleted (deleted thường đi kèm deleted_at)
  status          text not null default 'active',
  deleted_at      timestamptz,

  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now(),
  last_message_at timestamptz,

  constraint conversations_status_chk
    check (status in ('active','archived','deleted')),

  constraint conversations_deleted_chk
    check (
      (status <> 'deleted' and deleted_at is null)
      or
      (status = 'deleted' and deleted_at is not null)
    )
);

create index idx_conversations_user_updated
  on conversations(user_id, updated_at desc);

create index idx_conversations_user_lastmsg
  on conversations(user_id, last_message_at desc);

create index idx_conversations_user_active_lastmsg
  on conversations(user_id, last_message_at desc)
  where status <> 'deleted';

create table messages_30012026 (
  id              uuid primary key default gen_random_uuid(),
  conversation_id uuid not null references conversations(id) on delete cascade,
  user_id         integer not null references users(id) on delete cascade,

  role            text not null,
  content         text not null default '',
  content_json    jsonb not null default '{}'::jsonb,

  parent_id       uuid references messages(id) on delete set null,

  model           text,
  status          text not null default 'completed', -- pending|streaming|completed|error
  error_code      text,

  token_input     int,
  token_output    int,

  created_at      timestamptz not null default now(),
  updated_at      timestamptz not null default now(),

  constraint messages_role_chk
    check (role in ('user','assistant','system','tool')),

  constraint messages_status_chk
    check (status in ('pending','streaming','completed','error'))
);

create index idx_messages_conv_created
  on messages(conversation_id, created_at asc);

create index idx_messages_user_created
  on messages(user_id, created_at desc);

create table message_reactions_30012026 (
  id         uuid primary key default gen_random_uuid(),

  message_id uuid not null references messages(id) on delete cascade,
  user_id    integer not null references users(id) on delete cascade,

  reaction   smallint not null, -- 1=like, 2=dislike

  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),

  constraint reactions_reaction_chk
    check (reaction in (1,2)),

  constraint reactions_unique_per_user
    unique (message_id, user_id)
);

create index idx_reactions_message
  on message_reactions(message_id);

create index idx_reactions_user_updated
  on message_reactions(user_id, updated_at desc);

create table message_reports_30012026 (
  id              uuid primary key default gen_random_uuid(),

  message_id      uuid not null references messages(id) on delete cascade,
  conversation_id uuid not null references conversations(id) on delete cascade,
  user_id         integer not null references users(id) on delete cascade,

  reason_code     int not null, -- 1 spam, 2 hate, 3 sexual, 4 self-harm, 5 violence, 6 other...
  reason_text     text,

  created_at      timestamptz not null default now(),

  constraint reports_reason_chk
    check (reason_code in (1,2,3,4,5,6))
);

create index idx_reports_message
  on message_reports(message_id);

create index idx_reports_user_created
  on message_reports(user_id, created_at desc);

create index idx_reports_conv_created
  on message_reports(conversation_id, created_at desc);
